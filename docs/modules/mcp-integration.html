MCP Integration

Model Context Protocol (MCP) integration allows you to extend MAG SDK capabilities with external tools and services. Connect to databases, APIs, file systems, and custom services seamlessly.

Understanding MCP

Model Context Protocol (MCP) is a standardized way to connect AI models with external tools and data sources, enabling powerful integrations and extended functionality.

External Tools

Connect to file systems, databases, web APIs, and custom services

Real-time Data

Access live data and perform real-time operations during graph execution

Extensibility

Build custom MCP servers to extend functionality for specific use cases

Managing MCP Servers

Add, configure, and manage MCP servers in your MAG SDK environment:

Adding MCP Servers

# Add MCP servers
import mag

servers_config = {
    "filesystem": {
        "command": "npx",
        "args": ["-y", "@modelcontextprotocol/server-filesystem", "/path/to/files"],
        "description": "File system operations"
    },
    "database": {
        "command": "python",
        "args": ["-m", "mcp_server_sqlite", "--db-path", "./data.db"],
        "description": "SQLite database access"
    }
}

result = mag.add_mcp(servers_config)

if result.get("success"):
    print(f"Added MCP servers: {list(servers_config.keys())}")

    # Check server status
    status = mag.mcp_status()
    for server_name, server_info in status.items():
        print(f"{server_name}: {server_info.get('status', 'unknown')}")
else:
    print(f"Failed to add servers: {result.get('error')}")

Working with MCP Tools

Discover and use tools provided by MCP servers:

Discovering Available Tools

# Get all available MCP tools
tools = mag.list_mcp_tools()

print(f"Available MCP tools: {len(tools)}")
for tool in tools:
    server_name = tool.get("server")
    tool_name = tool.get("name")
    description = tool.get("description", "No description")

    print(f"\nðŸ“¦ {server_name}::{tool_name}")
    print(f"   {description}")

    # Show parameters if available
    params = tool.get("parameters", {})
    if params:
        print("   Parameters:")
        for param_name, param_info in params.items():
            required = "(required)" if param_info.get("required") else "(optional)"
            param_type = param_info.get("type", "unknown")
            print(f"     - {param_name}: {param_type} {required}")

Using MCP Tools in Graphs

# Register MCP tools for use in graphs
result = mag.register_mcp_tools(
    tools=[
        {
            "server": "filesystem",
            "name": "read_file",
            "alias": "read_file"  # Optional: custom alias
        },
        {
            "server": "filesystem",
            "name": "write_file",
            "alias": "save_file"
        },
        {
            "server": "database",
            "name": "query",
            "alias": "db_query"
        }
    ]
)

if result.get("success"):
    print(f"Registered {len(result.get('registered', []))} tools")
    print("Tools can now be used in graph node prompts")

    # Example: These tools are now available in your prompts
    print("\nExample usage in node prompts:")
    print("- Use read_file(path='data.txt') to read files")
    print("- Use save_file(path='output.txt', content='...')")
    print("- Use db_query(sql='SELECT * FROM users')")

File Processing with MCP

result = mag.create_graph(
    name="file_processor",
    nodes=[
        {
            "name": "reader",
            "prompt": "Read the input file using read_file(path={input}) and prepare the content for processing.",
            "next_node": "processor"
        },
        {
            "name": "processor",
            "prompt": "Process the file content: extract key information, clean data, and format results.",
            "next_node": "writer"
        },
        {
            "name": "writer",
            "prompt": "Save the processed results using save_file(path='output.txt', content=processed_data)."
        }
    ]
)

# Execute the file processing graph

    process_result = mag.run_graph(
        name="file_processor",

    )
    print(f"File processing result: {process_result.get('final_result')}")

Common MCP Servers

Filesystem Server

Access local file system operations

# Install
npm install -g @modelcontextprotocol/server-filesystem

# Add to MAG
mag.add_mcp_server(
    name="fs",
    command="npx",
    args=["-y", "@modelcontextprotocol/server-filesystem", "/allowed/path"]
)
Tools: read_file, write_file, list_directory, create_directory

Building Custom MCP Servers

Create your own MCP servers for specialized functionality:

# Generate a custom MCP server
result = mag.generate_mcp_server(
    description="Create a weather service MCP server with tools for:"
                "- Getting current weather for a location"
                "- Getting weather forecast"
                "- Weather alerts and warnings",
    server_name="weather_service",
    tools=[
        {
            "name": "get_weather",
            "description": "Get current weather for a location",
            "parameters": {
                "location": {"type": "string", "required": True},
                "units": {"type": "string", "default": "celsius"}
            }
        },
        {
            "name": "get_forecast",
            "description": "Get weather forecast",
            "parameters": {
                "location": {"type": "string", "required": True},
                "days": {"type": "integer", "default": 5}
            }
        }
    ]
)

if result.get("success"):
    print(f"Custom MCP server created: {result.get('server_path')}")
    print("You can now add it to MAG SDK:")

    add_result = mag.add_mcp_server(
        name="weather",
        command="python",
        args=[result.get("server_path")],
        description="Weather service integration"
    )
